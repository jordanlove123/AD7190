#include <SPI.h>
#include <AD7190.h>

AD7190 ad7190(2.5, true);
double v;

void setup()
{
  Serial.begin(9600);

  ad7190.init(AD7190::AIN1_2 | AD7190::AIN3_4);
}

void loop()
{
  v = ad7190.read_data(AD7190::AIN1_2);
  Serial.print(v, 6);

  v = ad7190.read_data(AD7190::AIN3_4);
  Serial.print(" : ");
  Serial.println(v, 6);
}
